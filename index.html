ปีเดือนวัน/เวลาอสม.ผู้คัดกรองหมู่ชื่อ-สกุลเพศวันเกิด (พ.ศ.)อายุเลขบัตรเบอร์โทรน้ำหนักส่วนสูงBMIรอบเอวการศึกษาอาชีพสถานภาพความดัน1 (SYS/DIA/ชีพจร)ความดัน2 (SYS/DIA/ชีพจร)ความดัน3 (SYS/DIA/ชีพจร)ค่าความดันเฉลี่ยกลุ่มความดันน้ำตาลค่าน้ำตาลกลุ่มน้ำตาลสูบบุหรี่มวน/วันชนิดบุหรี่ระยะเวลาสูบแอลกอฮอล์วัน/สัปดาห์ภาวะซึมเศร้า

👶 คัดกรองโภชนาการเด็ก 0-5 ปี

บันทึกและประเมินภาวะโภชนาการเด็กอายุ 0-5 ปี ตามเกณฑ์ WHO 2006

⚙️ ครั้งแรก: รัน SQL ใน Supabase เพื่อสร้างตาราง child_nutrition

CREATE TABLE IF NOT EXISTS public.child_nutrition (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  year integer, month integer,
  screen_date date, screen_time text,
  village integer, house_number text,
  full_name text NOT NULL,
  gender text, birthdate date,
  age_months integer,
  weight numeric, height numeric, head_circumference numeric,
  weight_status text, height_status text,
  weight_height_status text, head_circ_status text,
  overall_assessment text,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE public.child_nutrition ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON public.child_nutrition FOR ALL USING (true) WITH CHECK (true);
ปีเดือนวัน/เวลา หมู่บ้านเลขที่ชื่อ-สกุล เพศวันเกิดอายุ (เดือน) น้ำหนัก (กก.)ส่วนสูง (ซม.)รอบศีรษะ (ซม.) น้ำหนักตามเกณฑ์ส่วนสูงตามเกณฑ์ น้ำหนัก/ส่วนสูงรอบศีรษะ ผลรวม

🔬 คัดกรองมะเร็งปากมดลูก อายุ 30-60 ปี
